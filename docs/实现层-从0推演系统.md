```

逐步累加，从少到多

```


#### 最小实现

```python
# GpuAllocator
class GpuAllocator:
    def __init__(self, total_size: int):
        self.total_size = total_size
        self.used = 0
        self.free_blocks = [total_size]
    
    def allocate(self, size: int) -> Optional[int]:
        for i, block_size in enumerate(self.free_blocks):
            if block_size >= size:
                self.free_blocks.pop(i)
                self.used += size
                if block_size > size:
                    self.free_blocks.append(block_size - size)
                return size
        return None  # OOM
    
    def deallocate(self, size: int):
        self.used -= size
        self.free_blocks.append(size)
        self.free_blocks.sort()

# KV Cache
class KVCache:
    def __init__(self, allocator: GpuAllocator, max_blocks: int, block_size: int):
        self.block_size = block_size
        self.blocks = [allocator.allocate(block_size) for _ in range(max_blocks)]
        self.free_blocks = set(range(max_blocks))
    
    def allocate_block(self) -> Optional[int]:
        if not self.free_blocks:
            return None
        return self.free_blocks.pop()
    
    def free_block(self, block_id: int):
        self.free_blocks.add(block_id)

# KVCacheManager
class KVCacheManager:
    def __init__(self, total_size: int, block_size: int):
        self.allocator = GpuAllocator(total_size)
        self.cache = KVCache(self.allocator, total_size // block_size, block_size)
    
    def get_blocks(self, num_blocks: int) -> List[int]:
        blocks = []
        for _ in range(num_blocks):
            block = self.cache.allocate_block()
            if block is None:
                # 触发驱逐
                self._evict()
                block = self.cache.allocate_block()
                if block is None:
                    raise VllmError("OOM")
            blocks.append(block)
        return blocks
    
    def _evict(self):
        # LRU 驱逐策略
        pass
```

#### 推演问题

- 为什么要先有 GpuAllocator？因为显存是有限资源
- KV Cache 和普通缓存的区别？KV Cache 是注意力计算的中间结果，需要按需分配
- 为什么需要 KVCacheManager？因为多个请求共享缓存，需要调度

#### PagedAttention 最小实现（核心）

这是 vLLM 最核心的创新，需要重点理解：

```python
# Block Table（块表）
class BlockTable:
    def __init__(self, max_blocks: int):
        self.table = {}  # token_id -> list of block_ids
        self.max_blocks = max_blocks
    
    def add_token(self, token_id: int, block_id: int):
        if token_id not in self.table:
            self.table[token_id] = []
        self.table[token_id].append(block_id)
    
    def get_blocks(self, token_ids: List[int]) -> List[int]:
        blocks = []
        for tid in token_ids:
            blocks.extend(self.table.get(tid, []))
        return blocks

# PagedAttention
class PagedAttention:
    def __init__(self, cache_manager: KVCacheManager):
        self.cache_manager = cache_manager
        self.block_size = 16
    
    def forward(
        self,
        query: Tensor,
        key: Tensor,
        value: Tensor,
        block_table: BlockTable,
        position_ids: List[int]
    ) -> Tensor:
        # 1. 计算 query 在哪些 block
        block_ids = [pos // self.block_size for pos in position_ids]
        
        # 2. 通过 block_table 查找物理块
        physical_blocks = block_table.get_blocks(block_ids)
        
        # 3. 从 KV Cache 读取数据
        k_cache = self.cache_manager.get_k_cache(physical_blocks)
        v_cache = self.cache_manager.get_v_cache(physical_blocks)
        
        # 4. 计算注意力（分块）
        output = self._compute_attention(query, k_cache, v_cache)
        
        # 5. 存储新的 KV 到 cache
        self.cache_manager.store_kv(position_ids[-1], key, value)
        
        return output
    
    def _compute_attention(self, q, k, v) -> Tensor:
        # 标准 attention 计算
        scale = 1.0 / math.sqrt(q.shape[-1])
        scores = torch.matmul(q, k.transpose(-2, -1)) * scale
        attn = torch.softmax(scores, dim=-1)
        return torch.matmul(attn, v)

# Sampler
class Sampler:
    def __init__(self):
        pass
    
    def sample(self, logits: Tensor, params: SamplingParams) -> int:
        if params.temperature == 0:
            # Greedy
            return torch.argmax(logits).item()
        
        # Temperature sampling
        probs = torch.softmax(logits / params.temperature, dim=-1)
        
        if params.top_k > 0:
            top_k = min(params.top_k, probs.numel())
            values, indices = torch.topk(probs, top_k)
            probs = torch.zeros_like(probs)
            probs[indices] = values
        
        if params.top_p > 0:
            # Nucleus sampling
            sorted_probs, sorted_indices = torch.sort(probs, descending=True)
            cumsum = torch.cumsum(sorted_probs, dim=-1)
            mask = cumsum > params.top_p
            sorted_probs[mask] = 0
            probs[sorted_indices] = sorted_probs
            probs = probs / probs.sum()
        
        return torch.multinomial(probs, 1).item()
```

#### 推演问题

- 为什么需要 PagedAttention？因为传统方法需要预分配固定长度的 KV Cache，导致碎片化
- Block Table 解决什么问题？解决虚拟块到物理块的映射问题
- 为什么 PagedAttention 快？按需分配，避免内存浪费，减少分配开销

```python
# Logits（模型输出）
# Logits 是模型最后的输出层，维度是 vocab_size

# Token
class Tokenizer:
    def __init__(self, vocab_file: str):
        self.vocab = self._load_vocab(vocab_file)
        self.reverse_vocab = {v: k for k, v in self.vocab.items()}
    
    def encode(self, text: str) -> List[int]:
        # 分词
        return [self.vocab.get(t, self.vocab['<unk>']) for t in text]
    
    def decode(self, ids: List[int]) -> str:
        return ''.join([self.reverse_vocab[i] for i in ids])

# Decode Step
class DecodeStep:
    def __init__(self, model_runner: ModelRunner, sampler: Sampler):
        self.model_runner = model_runner
        self.sampler = sampler
    
    def step(self, input_ids: List[int], params: SamplingParams) -> int:
        # 1. 转换为 tensor
        input_tensor = Tensor((len(input_ids),), device)
        
        # 2. 前向计算
        logits = self.model_runner.forward(input_tensor)
        
        # 3. 采样
        next_token = self.sampler.sample(logits, params)
        
        return next_token
```

```python
# Request
class Request:
    def __init__(self, request_id: str, prompt: List[int], params: SamplingParams):
        self.request_id = request_id
        self.prompt = prompt
        self.params = params
        self.output_ids = []
        self.status = "waiting"  # waiting, running, completed, failed

# Request Queue
class RequestQueue:
    def __init__(self, max_size: int = 100):
        self.queue = deque()
        self.max_size = max_size
    
    def add(self, request: Request):
        if len(self.queue) >= self.max_size:
            raise VllmError("Queue full")
        self.queue.append(request)
    
    def pop(self) -> Request:
        return self.queue.popleft()
    
    def is_empty(self) -> bool:
        return len(self.queue) == 0

# Scheduler + Continuous Batching
class Scheduler:
    def __init__(self, queue: RequestQueue, max_batch_size: int):
        self.queue = queue
        self.max_batch_size = max_batch_size
        self.running = []
    
    def schedule(self) -> List[Request]:
        # 1. 从队列中取出请求直到满 batch
        batch = []
        while len(batch) < self.max_batch_size and not self.queue.is_empty():
            req = self.queue.pop()
            req.status = "running"
            batch.append(req)
        
        # 2. 将正在运行的请求中已完成的移除
        self.running = [r for r in self.running if r.status != "completed"]
        
        return batch
```

| 24 | Decode Step | 解码循环 |

### 最小代码量

```python
# 最小 vLLM 原型（概念性）

class MinimalVllm:
    def __init__(self):
        self.device = Device("cuda")
        self.allocator = GpuAllocator(80 * 1024**3)  # 80GB
        self.cache = KVCache(self.allocator, block_size=16)
        self.model = Model()
        self.sampler = Sampler()
        self.scheduler = Scheduler()
    
    def generate(self, prompt: str) -> str:
        # 简化的生成流程
        input_ids = self._tokenize(prompt)
        
        for _ in range(100):  # max_tokens
            # 1. 前向计算
            logits = self.model.forward(input_ids)
            
            # 2. 采样
            next_token = self.sampler.sample(logits, temperature=0)
            
            # 3. 添加到输出
            input_ids.append(next_token)
            
            if next_token == self.eos_id:
                break
        
        return self._detokenize(input_ids)
```
